create table contas ( 
    id BIGINT NOT NULL AUTO_INCREMENT,
    titular varchar(50),
    cpf varchar(11),
    email varchar(60),
    telefone varchar(12),
    celular varchar(14),
    endereco varchar(60),
    complemento varchar(20),
    numero_residencial varchar(10),
    pais varchar(40),
    estado varchar(50),
    cep varchar(15),
    tipo_conta varchar(60),
    numero varchar(16),
    senha varchar(6),
    saldo double, 
    limite double,
    primary key(id));
